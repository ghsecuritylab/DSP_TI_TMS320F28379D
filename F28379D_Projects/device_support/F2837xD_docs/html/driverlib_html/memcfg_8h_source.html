265 static inline void
266 MemCfg_setCLAMemType(uint32_t ramSections, MemCfg_CLAMemoryType claMemType)
267 {
268  //
269  // Check the arguments.
270  //
271  ASSERT((ramSections & MEMCFG_SECT_TYPE_MASK) == MEMCFG_SECT_TYPE_LS);
272 
273  //
274  // Write the CLA memory configuration to the appropriate register. Either
275  // set or clear the bit that determines the function of the RAM section as
276  // it relates to the CLA.
277  //
278  EALLOW;
279 
280  if(claMemType == MEMCFG_CLA_MEM_PROGRAM)
281  {
282  // Program memory
283  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) |= ramSections;
284  }
285  else
286  {
287  // Data memory
288  HWREG(MEMCFG_BASE + MEMCFG_O_LSXCLAPGM) &= ~ramSections;
289  }
290 
291  EDIS;
292 }

319 MemCfg_enableViolationInterrupt(uint32_t intFlags)

321  //
322  // Enable the specified interrupts.

403 {
404  uint32_t status;

407  // Read and return RAM access status flags.
408  //

411  MEMCFG_MVIOL_SHIFT);
412 
413  return(status);
