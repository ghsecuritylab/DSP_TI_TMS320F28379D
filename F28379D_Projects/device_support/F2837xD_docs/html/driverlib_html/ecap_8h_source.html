205 static inline void ECAP_setEventPrescaler(uint32_t base,
206  uint16_t preScalerValue)
207 {
208  ASSERT(ECAP_isBaseValid(base));
209 
210  ASSERT(preScalerValue < ECAP_MAX_PRESCALER_VALUE);
211 
212 
213  //
214  // Write to PRESCALE bit
215  //
216  HWREGH(base + ECAP_O_ECCTL1) =
217  ((HWREGH(base + ECAP_O_ECCTL1) & (~ECAP_ECCTL1_PRESCALE_M)) |
218  (preScalerValue << ECAP_ECCTL1_PRESCALE_S));
219 }

240  ECAP_Events event,
241  ECAP_EventPolarity polarity)
242 {
243 
244  uint16_t shift;
245 
246  ASSERT(ECAP_isBaseValid(base));
247 
248  shift = ((uint16_t)event) << 1U;
249 
250 
251  //
252  // Write to CAP1POL, CAP2POL, CAP3POL or CAP4POL
253  //
254  HWREGH(base + ECAP_O_ECCTL1) =
255  (HWREGH(base + ECAP_O_ECCTL1) & ~(1U << shift)) |
256  ((uint16_t)polarity << shift);
257 }

1043 static inline uint32_t ECAP_getEventTimeStamp(uint32_t base, ECAP_Events event)
1044 {
1045  uint32_t count;
1046 
1047  ASSERT(ECAP_isBaseValid(base));
1048 
1049 
1050  switch(event)
1051  {
1052  case ECAP_EVENT_1:
1053 
1054  //
1055  // Read CAP1 register
1056  //
1057  count = HWREG(base + ECAP_O_CAP1);
1058  break;
1059 
1060  case ECAP_EVENT_2:
1061  //
1062  // Read CAP2 register
1063  //
1064  count = HWREG(base + ECAP_O_CAP2);
1065  break;
1066 
1067  case ECAP_EVENT_3:
1068 
1069  //
1070  // Read CAP3 register
1071  //
1072  count = HWREG(base + ECAP_O_CAP3);
1073  break;
1074 
1075  case ECAP_EVENT_4:
1076 
1077  //
1078  // Read CAP4 register
1079  //
1080  count = HWREG(base + ECAP_O_CAP4);
1081  break;
1082 
1083  default:
1084 
1085  //
1086  // Invalid event parameter
1087  //
1088  count = 0U;
1089  break;
1090  }
1091 
1092  return(count);
1093 }
