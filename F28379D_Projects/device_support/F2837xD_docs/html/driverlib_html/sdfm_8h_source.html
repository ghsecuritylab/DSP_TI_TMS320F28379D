612 //*****************************************************************************
613 static inline void SDFM_enableInterrupt(uint32_t base,
614  SDFM_FilterNumber filterNumber,
615  uint16_t intFlags)
616 {
617  uint16_t offset;
618 
619  ASSERT(SDFM_isBaseValid(base));
620 
621  offset = (uint16_t)filterNumber * 16U;
622 
623  EALLOW;
624 
625  //
626  // Low, high threshold, Modulator failure
627  //
628  if((intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
629  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
630  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT)) != 0U)
631  {
632  //
633  // Set IEL or IEH or MFIE bit of SDFM_O_SDCPARMx
634  //
635  HWREGH(base + SDFM_O_SDCPARM1 + offset) |=
636  (intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
637  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
638  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT));
639  }
640 
641  //
642  // Data filter acknowledge interrupt
643  //
644  if((intFlags & SDFM_DATA_FILTER_ACKNOWLEDGE_INTERRUPT) != 0U)
645  {
646  HWREGH(base + SDFM_O_SDDFPARM1 + offset) |= SDFM_SDDFPARM1_AE;
647  }
648  EDIS;
649 }

669 static inline void SDFM_disableInterrupt(uint32_t base,
670  SDFM_FilterNumber filterNumber,
671  uint16_t intFlags)
672 {
673  uint16_t offset;
674 
675  ASSERT(SDFM_isBaseValid(base));
676 
677  offset = (uint16_t)filterNumber * 16U;
678 
679  EALLOW;
680 
681  //
682  // Low, high threshold, modulator failure interrupts
683  //
684  if((intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
685  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
686  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT)) != 0U)
687  {
688  //
689  // Set IEL or IEH or MFIE bit of SDFM_O_SDCPARMx
690  //
691  HWREGH(base + SDFM_O_SDCPARM1 + offset) &=
692  ~(intFlags & (SDFM_MODULATOR_FAILURE_INTERRUPT |
693  SDFM_LOW_LEVEL_THRESHOLD_INTERRUPT |
694  SDFM_HIGH_LEVEL_THRESHOLD_INTERRUPT));
695  }
696 
697  //
698  // Data filter acknowledge interrupt
699  //
700  if((intFlags & SDFM_DATA_FILTER_ACKNOWLEDGE_INTERRUPT) != 0U)
701  {
702  HWREGH(base + SDFM_O_SDDFPARM1 + offset) &= ~SDFM_SDDFPARM1_AE;
703  }
704  EDIS;
705 }
