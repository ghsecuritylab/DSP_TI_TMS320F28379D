428 ADC_setupSOC(uint32_t base, ADC_SOCNumber socNumber, ADC_Trigger trigger,
429  ADC_Channel channel, uint32_t sampleWindow)
430 {
431  uint32_t ctlRegAddr;
432 
433  //
434  // Check the arguments.
435  //
436  ASSERT(ADC_isBaseValid(base));
437  ASSERT((sampleWindow >= 1U) && (sampleWindow <= 512U));
438 
439  //
440  // Calculate address for the SOC control register.
441  //
442  ctlRegAddr = base + ADC_SOCxCTL_OFFSET_BASE + ((uint32_t)socNumber * 2U);
443 
444  //
445  // Set the configuration of the specified SOC.
446  //
447  EALLOW;
448  HWREG(ctlRegAddr) = ((uint32_t)channel << ADC_SOC0CTL_CHSEL_S) |
449  ((uint32_t)trigger << ADC_SOC0CTL_TRIGSEL_S) |
450  (sampleWindow - 1U);
451  EDIS;
452 }
