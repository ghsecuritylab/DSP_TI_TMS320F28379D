222 DMA_initController(void)
223 {
224  EALLOW;
225 
226  //
227  // Set the hard reset bit. One NOP is required after HARDRESET.
228  //
229  HWREGH(DMA_BASE + DMA_O_CTRL) |= DMA_CTRL_HARDRESET;
230  NOP;
231 
232  EDIS;
233 }
