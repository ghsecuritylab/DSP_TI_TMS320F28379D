252 static inline DCSM_SecurityStatus
253 DCSM_getZone1CSMSecurityStatus(void)
254 {
255  uint16_t status;
256  DCSM_SecurityStatus returnStatus;
257  status = HWREGH(DCSM_Z1_BASE + DCSM_O_Z1_CR);
258 
259  //
260  // if ARMED bit is set and UNSECURED bit or ALLONE bit or both UNSECURED
261  // and ALLONE bits are set then CSM is unsecured. Else it is secure.
262  //
263  if(((status & DCSM_Z1_CR_ARMED) != 0U) &&
264  (((status & DCSM_Z1_CR_UNSECURE) != 0U) ||
265  ((status & DCSM_Z1_CR_ALLONE) != 0U )))
266  {
267  returnStatus = DCSM_STATUS_UNSECURE;
268  }
269  else if((status & DCSM_Z1_CR_ALLZERO) == DCSM_Z1_CR_ALLZERO)
270  {
271  returnStatus = DCSM_STATUS_LOCKED;
272  }
273  else
274  {
275  returnStatus = DCSM_STATUS_SECURE;
276  }
277 
278  return(returnStatus);
279 }
