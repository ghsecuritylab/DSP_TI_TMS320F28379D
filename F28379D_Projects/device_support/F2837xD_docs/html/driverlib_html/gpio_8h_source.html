358 static inline uint32_t
359 GPIO_readPin(uint32_t pin)
360 {
361  volatile uint32_t *gpioDataReg;
362 
363  //
364  // Check the arguments.
365  //
366  ASSERT(GPIO_isPinValid(pin));
367 
368  gpioDataReg = (uint32_t *)GPIODATA_BASE +
369  ((pin / 32U) * GPIO_DATA_REGS_STEP);
370 
371  return((gpioDataReg[GPIO_GPxDAT_INDEX] >> (pin % 32U)) & (uint32_t)0x1U);
372 }
